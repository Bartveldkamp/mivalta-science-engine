# TRAIN v5: Humanized dual-model (Interpreter + Explainer)
# ============================================================
# Data fixes applied:
#   - 529/1120 explainer examples rewritten (robot speak, jargon, templates)
#   - 8 interpreter mislabels fixed (Z2→Z5 was "reduce_intensity")
#   - System prompt updated (technical terms OK when explained)
#
# Two models trained separately:
#   - Interpreter: user message → GATCRequest JSON
#   - Explainer:   GATC context → plain coaching text

# ── STEP 1: Clone repo on GPU server ──────────────────────
ssh root@<YOUR-GPU-SERVER>
mkdir -p ~/mivalta && cd ~/mivalta
git clone git@github.com:Bartveldkamp/mivalta-science-engine.git
cd mivalta-science-engine
git checkout claude/review-llm-implementations-HGMMa

# ── STEP 2: Setup (installs PyTorch, deps, downloads model) ──
cd training
bash scripts/setup_gpu.sh

# ── STEP 3: Train both models ────────────────────────────
source .venv/bin/activate
bash scripts/train_v5.sh              # Trains interpreter + explainer
# OR: bash scripts/train_v5.sh interpreter   # Interpreter only
# OR: bash scripts/train_v5.sh explainer     # Explainer only

# ── STEP 4: Check results ────────────────────────────────
ls -la models/josi-v*

# ── STEP 5: Merge LoRA weights ───────────────────────────
# (Replace FOLDER with actual folder name from step 4)
python scripts/finetune_gemma3n.py merge \
    --lora_path ./models/FOLDER/lora_weights \
    --output_path ./models/josi-v5-interpreter-merged

python scripts/finetune_gemma3n.py merge \
    --lora_path ./models/FOLDER/lora_weights \
    --output_path ./models/josi-v5-explainer-merged

# ── STEP 6: Export to GGUF for on-device ─────────────────
python scripts/export_gguf.py --model ./models/josi-v5-interpreter-merged
python scripts/export_gguf.py --model ./models/josi-v5-explainer-merged

# ── STEP 7: Evaluate ─────────────────────────────────────
python scripts/evaluate_gemma3n_v4.py --model ./models/josi-v5-interpreter-merged --mode interpreter
python scripts/evaluate_gemma3n_v4.py --model ./models/josi-v5-explainer-merged --mode explainer

# ── STEP 8: Share GGUF with developer ────────────────────
cd ~/mivalta-science-engine/training/models/gguf
nohup python3 -m http.server 8888 > /dev/null 2>&1 &
# Kill when done: kill $(lsof -t -i:8888)
